
\begin{ReqNoFun}
	\reqNoFunItem{RNF-001}{Rapidez}{La aplicación debe contar con un tiempo de respuesta mínimo dado que no consume recursos excesivos.}
    
    \reqNoFunItem{RNF-002}{Usabilidad}{La aplicación fue diseñada e implementada bajo la Guía de Interfaces Humanas de Apple que ayuda a diseñar una aplicación con altos estándares de calidad.}

	\reqNoFunItem{RNF-003}{Intuitivo}{El entorno de desarrollo permite hacer una aplicación sencilla para el usuario, ya sea de manejo como de actualización.}
    
    \reqNoFunItem{RNF-004}{Eficiente}{La ubicación de los espacios e información de los profesores debe ser correcta.}
    
    \reqNoFunItem{RNF-005}{Robustez}{Hasta el momento y durante la fase de prototipos, no se registraron incidencias en errores solucionados.}
    
    \reqNoFunItem{RNF-06}{Portabilidad}{El lenguaje e IDE de programación permite la integración de las distintas versiones de dispositivos que ofrece Apple.}
    
    \reqNoFunItem{RNF-007}{Equipo de cómputo}{El desarrollo de la aplicación debe ser realizada desde una computadora de la marca Apple debido a sus restricciones de desarrollo.}
    
    \reqNoFunItem{RNF-008}{Sistema Operativo MacOS}{El desarrollo de la aplicación debe ser realizada desde una computadora con sistema operativo MacOS debido a sus restrcciones de desarrollo.}

	\reqNoFunItem{RNF-009}{API de MapKit}{Herramienta para el uso de Mapas de la empresa Apple.}
	
	\reqNoFunItem{RNF-010}{iPhone}{Teléfono inteligente de la marca Apple específicamente el modelo 7 plus.}
	
	\reqNoFunItem{RNF-011}{Sistema Operativo iOS}{El desarrollo corre bajo el sistema operativo nativo para los teléfonos de la marca Apple}
	
	\reqNoFunItem{RNF-012}{Lenguaje de programación}{Se desarrolló en lenguaje dado Swift ya que éste es uno de los lenguajes oficiales de programación de aplicaciones móviles para iOS, pero el único que cumple con los requerimientos necesarios para el desarrollo.}
	
	 \reqNoFunItem{RNF-013}{Entorno de Desarrollo}{La aplicación será desarrollada utilizando el entorno de desarrollo de la empresa Apple llamado XCode.}
	 
	 \reqNoFunItem{RNF-014}{Servidor}{El servidor que alojará los servicios web propuestos en la segunda fase de Trabajo Terminal será Tomcat.}
\end{ReqNoFun}
