\begin{UseCase}{CUI 2}{Consultar indicadores de agua}
	{
		Este caso de uso permitirá al actor consultar los indicadores ambientales y de sustentabilidad de la línea de acción ``Agua'', 
		estos indicadores permitirán medir, monitorear, evaluar y comunicar el impacto del Programa de Acreditación de Escuelas Ambientalmente Responsables.
	}
	%\UCitem{\DONEUC}{Edición}
	\UCitem{Versión}{1.0}
	\UCccsection{Administración de Requerimientos}	
	\UCitem{Autor}{Sergio Ramírez Camaho}	
	\UCccitem{Evaluador}{}
	\UCitem{Operación}{Consulta}
	\UCccitem{Prioridad}{Alta}
	\UCccitem{Complejidad}{Media}
	\UCccitem{Volatilidad}{Alta}
	\UCccitem{Madurez}{Baja}
	\UCitem{Estatus}{Terminado}
	\UCitem{Fecha del último estatus}{9 de diciembre de 2014}

	\UCsection{Atributos}
	\UCitem{Actor}{\cdtRef{actor:usuarioEscuela}{Coordinador del programa}}
	\UCitem{Propósito}{
		Consultar los indicadores ambientales y de sustentabilidad de la línea de acción ``Agua''.
	}
	\UCitem{Entradas}{
		Ninguna
	}
	\UCitem{Salidas}{
		\begin{UClist}
			\UCli \cdtRef{gls:consumoAnual}{Consumo anual de agua}: \ioCalcularIndicador \cdtIdRef{IA 1}{Consumo anual de agua}.
			\UCli \cdtRef{gls:consumoAnualPersona}{Consumo anual por persona}: \ioCalcularIndicador \cdtIdRef{IA 2}{Consumo anual de agua por persona}.
			\UCli \cdtRef{gls:disminucionAnual}{Disminución anual del consumo de agua}: \ioCalcularIndicador \cdtIdRef{IA 3}{Disminución anual en consumo de agua}.
			\UCli \cdtRef{gls:disminucionEmisiones}{Disminución de emisiones de $CO_2$}: \ioCalcularIndicador \cdtIdRef{IA 4}{Disminución de emisiones a la atmósfera por reducción en consumo de agua por año}.		
			\UCli \cdtIdRef{MSG42}{No puede calcularse el indicador}: Se muestra en la pantalla \cdtIdRef{IUI 2}{Consultar indicadores de agua} indicando que no se puede realizar el cálculo del valor del indicador
				     debido a que no se cuenta con la información necesaria.
		\end{UClist}		
	}

	\UCitem{Precondiciones}{
		\begin{UClist}
			\UCli {\bf Interna:} Que el estado de la escuela sea \cdtRef{estado:porAcreditar}{Por acreditar}.
		\end{UClist}
	
	}
	
	\UCitem{Postcondiciones}{
		Ninguna
	}

	\UCitem{Reglas de \hspace{1 cm} negocio}{
		Ninguna
	}

	\UCitem{Errores}{
		\begin{UClist}	
			\UCli \cdtIdRef{MSG28}{Operación no permitida por estado de la entidad}: Se muestra en la pantalla en que se encuentre navegando el actor cuando el estado de la escuela no es ``Por acreditar''.
		\end{UClist}
	}

	
	\UCitem{Tipo}{Secundario, extiende del caso de uso \cdtIdRef{CUI 1}{Consultar indicadores}}

%-------------------------------------------------------%trayectoria Principal-----------------------------------------------
 \begin{UCtrayectoria}
    \UCpaso[\UCactor] Solicita consultar los indicadores de agua seleccionando la pestaña ``Agua'' en la pantalla \cdtIdRef{IUI 1}{Consultar indicadores}.
    \UCpaso[\UCsist] Verifica que el estado de la escuela sea ``Por acreditar''. \refTray{A}
    \UCpaso[\UCsist] Calcula el valor de los indicadores de  agua tomando la información más reciente de la escuela.%, como lo indica la regla de negocio \cdtRef{RN-18}{Información para indicadores}.
    \UCpaso[\UCsist] Muestra la pantalla \cdtIdRef{IUI 2}{Consultar indicadores de agua} en la cual el actor podrá visualizar la información de los indicadores de agua.
\end{UCtrayectoria}
 

 \begin{UCtrayectoriaA}[Fin del caso de uso]{A}{La escuela no se encuentra en el estado ``Por acreditar''.}
    \UCpaso[\UCsist] Muestra el mensaje \cdtIdRef{MSG28}{Operación no permitida por estado de la entidad} en la pantalla en que se encuentre navegando el
    actor indicando que no puede consultar los indicadores debido a que la escuela no se encuentra en el estado ``Por acreditar''. 
 \end{UCtrayectoriaA}
