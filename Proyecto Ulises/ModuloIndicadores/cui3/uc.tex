\begin{UseCase}{CUI 3}{Consultar indicadores de residuos sólidos}
	{
		Este caso de uso permitirá al actor consultar los indicadores ambientales y de sustentabilidad de la línea de acción ``Residuos sólidos'', 
		estos indicadores permitirán medir, monitorear, evaluar y comunicar el impacto del Programa de Acreditación de Escuelas Ambientalmente Responsables.
	}

	\UCitem{Versión}{1.0}

	\UCitem{Autor}{Sergio Ramírez Camaho}

	\UCitem{Operación}{Consulta}

	\UCccitem{Complejidad}{Media}
	\UCccitem{Volatilidad}{Alta}
	\UCccitem{Madurez}{Baja}
	\UCitem{Estatus}{Terminado}
	\UCitem{Fecha del último estatus}{9 de diciembre de 2014}

	\UCsection{Atributos}
	\UCitem{Actor}{\cdtRef{actor:usuarioEscuela}{Coordinador del programa}}
	\UCitem{Propósito}{
		Consultar los indicadores ambientales y de sustentabilidad de la línea de acción ``Residuos sólidos''.
	}
	\UCitem{Entradas}{
		Ninguna
	}
	\UCitem{Salidas}{
		\begin{UClist}
 			\UCli \cdtRef{gls:residuosAno}{Residuos sólidos generados al año por persona}: \ioCalcularIndicador \cdtIdRef{IR 1}{Residuos generados por persona en la escuela por año}.
  			\UCli \cdtRef{gls:disminucionRSAnoPersona}{Disminución de residuos sólidos generados al año por persona}: \ioCalcularIndicador \cdtIdRef{IR 2}{Disminución de residuos por persona al año}.
  			\UCli Porcentaje de la \cdtRef{gls:disminucionRSAnoPersona}{disminución de residuos sólidos generados al año por persona}: \ioCalcularIndicador \cdtIdRef{IR 3}{Porcentaje de disminución de residuos por persona al año}.
  			\UCli \cdtRef{gls:residuosReciclaje}{Residuos enviados a reciclaje por año}: \ioCalcularIndicador \cdtIdRef{IR 4}{Cantidad de residuos en kg enviados a reciclaje}.					\UCli \cdtRef{gls:disminucionEmisiones}{Disminución de emisiones de $CO_2$}: \ioCalcularIndicador \cdtIdRef{IR 5}{Disminución de emisiones por reciclaje de papel}.		
  			\UCli \cdtIdRef{MSG42}{No puede calcularse el indicador}: Se muestra en la pantalla \cdtIdRef{IUI 3}{Consultar indicadores de residuos sólidos} indicando que no se puede realizar el cálculo del valor del indicador
  				     debido a que no se cuenta con la información necesaria.
		\end{UClist}		
	}

	\UCitem{Precondiciones}{
	  	\begin{UClist}
			\UCli {\bf Interna:} Que el estado de la escuela sea \cdtRef{estado:porAcreditar}{Por acreditar}.
		\end{UClist}
		
	}
	
	\UCitem{Postcondiciones}{
		Ninguna

		\end{UClist}
	}

%-------------------------------------------------------%trayectoria Principal-----------------------------------------------
 \begin{UCtrayectoria}
    \UCpaso[\UCactor] Solicita consultar los indicadores de residuos sólidos seleccionando la pestaña ``Residuos sólidos'' en la pantalla \cdtIdRef{IUI 1}{Consultar indicadores}.
    \UCpaso[\UCsist] Verifica que el estado de la escuela sea ``Por acreditar''. \refTray{A}
    \UCpaso[\UCsist] Calcula el valor de los indicadores de residuos sólidos tomando la información más reciente de la escuela.%, como lo indica la regla de negocio \cdtRef{RN-18}{Información para indicadores}.
    \UCpaso[\UCsist] Muestra la pantalla \cdtIdRef{IUI 3}{Consultar indicadores de residuos sólidos} en la cual el actor podrá visualizar la información de los indicadores de residuos sólidos.
\end{UCtrayectoria}
 
 \begin{UCtrayectoriaA}[Fin del caso de uso]{A}{La escuela no se encuentra en el estado ``Por acreditar''.}
    \UCpaso[\UCsist] Muestra el mensaje \cdtIdRef{MSG28}{Operación no permitida por estado de la entidad} en la pantalla en que se encuentre navegando el
    actor indicando que no puede consultar los indicadores debido a que la escuela no se encuentra en el estado ``Por acreditar''. 
 \end{UCtrayectoriaA}
