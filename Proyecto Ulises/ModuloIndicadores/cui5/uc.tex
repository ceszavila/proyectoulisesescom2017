\begin{UseCase}{CUI 5}{Consultar indicadores de biodiversidad}
	{
		Este caso de uso permitirá al actor consultar los indicadores ambientales y de sustentabilidad de la línea de acción ``Biodiversidad'', 
		estos indicadores permitirán medir, monitorear, evaluar y comunicar el impacto del Programa de Acreditación de Escuelas Ambientalmente Responsables.
	}
	%\UCitem{\DONEUC}{Edición}
	\UCitem{Versión}{1.0}
	\UCccsection{Administración de Requerimientos}	
	\UCitem{Autor}{Sergio Ramírez Camaho}	
	\UCccitem{Evaluador}{}
	\UCitem{Operación}{Consulta}
	\UCccitem{Prioridad}{Alta}
	\UCccitem{Complejidad}{Media}
	\UCccitem{Volatilidad}{Alta}
	\UCccitem{Madurez}{Baja}
	\UCitem{Estatus}{Terminado}
	\UCitem{Fecha del último estatus}{9 de diciembre de 2014}
%% Copie y pegue este bloque tantas veces como revisiones tenga el caso de uso.
%% Esta sección la debe llenar solo el Revisor

	\UCsection{Atributos}
	\UCitem{Actor}{\cdtRef{actor:usuarioEscuela}{Coordinador del programa}}
	\UCitem{Propósito}{
		Consultar los indicadores ambientales y de sustentabilidad de la línea de acción ``Biodiversidad''.
	}
	\UCitem{Entradas}{
		Ninguna
	}
	\UCitem{Salidas}{
 		\begin{UClist}
 			\UCli \cdtRef{gls:personasBio}{Porcentaje de personas que conocen el concepto de biodiversidad}: \ioCalcularIndicador \cdtIdRef{IB 1}{Porcentaje de personas que conocen el concepto de biodiversidad}.
  			\UCli \cdtRef{gls:totalEspecies}{Número total de especies}: \ioCalcularIndicador \cdtIdRef{IB 2}{Número total de especies de flora y fauna presentes en la escuela por año}.
  			\UCli \cdtRef{gls:porcentajeEspeciesEndemicas}{Porcentaje de especies endémicas}: \ioCalcularIndicador \cdtIdRef{IB 3}{Porcentaje de especies endémicas en la escuela por año}.
			\UCli \cdtRef{gls:arbolesPlantados}{Árboles plantados}: \ioCalcularIndicador \cdtIdRef{IB 4}{Número de árboles plantados en un año}.		
  			\UCli \cdtRef{gls:tasaSupervivencia}{Tasa de supervivencia de árboles plantados}: \ioCalcularIndicador \cdtIdRef{IB 5}{Tasa de supervivencia de árboles plantados}.					
  			\UCli \cdtRef{gls:numeroEco}{Número de ecosistemas identificados}: \ioCalcularIndicador \cdtIdRef{IB 6}{Número de ecosistemas locales}.		
  			\UCli \cdtRef{gls:porcentajeAV}{Porcentaje de áreas verdes}: \ioCalcularIndicador \cdtIdRef{IB 7}{Porcentaje de áreas verdes en la escuela}.		
  			\UCli \cdtRef{gls:aVPersona}{Superficie de áreas verdes por persona}: \ioCalcularIndicador \cdtIdRef{IB 8}{Metros cuadrados de áreas verdes por persona en la escuela}.					
  			\UCli \cdtIdRef{MSG42}{No puede calcularse el indicador}: Se muestra en la pantalla \cdtIdRef{IUI 5}{Consultar indicadores de biodiversidad} indicando que no se puede realizar el cálculo del valor del indicador
  				     debido a que no se cuenta con la información necesaria.
 		\end{UClist}		
	}

	\UCitem{Precondiciones}{
	  	\begin{UClist}
			\UCli {\bf Interna:} Que el estado de la escuela sea \cdtRef{estado:porAcreditar}{Por acreditar}.
		\end{UClist}
		
	}
	
	\UCitem{Postcondiciones}{
		Ninguna
	}

	\UCitem{Reglas de \hspace{1 cm} negocio}{
		Ninguna
	}

	\UCitem{Errores}{
		\begin{UClist}	
			\UCli \cdtIdRef{MSG28}{Operación no permitida por estado de la entidad}: Se muestra en la pantalla en que se encuentre navegando el actor cuando el estado de la escuela no es ``Por acreditar''.
		\end{UClist}
	}

	
	\UCitem{Tipo}{Secundario, extiende del caso de uso \cdtIdRef{CUI 1}{Consultar indicadores}}

\end{UseCase}
%-------------------------------------------------------%trayectoria Principal-----------------------------------------------
 \begin{UCtrayectoria}
    \UCpaso[\UCactor] Solicita consultar los indicadores de biodiversidad seleccionando la pestaña ``Biodiversidad'' en la pantalla \cdtIdRef{IUI 1}{Consultar indicadores}.
    \UCpaso[\UCsist] Verifica que el estado de la escuela sea ``Por acreditar''. \refTray{A}  
    \UCpaso[\UCsist] Calcula el valor de los indicadores de  biodiversidad tomando la información más reciente de la escuela.%, como lo indica la regla de negocio \cdtRef{RN-18}{Información para indicadores}.
    \UCpaso[\UCsist] Muestra la pantalla \cdtIdRef{IUI 5}{Consultar indicadores de biodiversidad} en la cual el actor podrá visualizar la información de los indicadores de biodiversidad.
\end{UCtrayectoria}
 
 \begin{UCtrayectoriaA}[Fin del caso de uso]{A}{La escuela no se encuentra en el estado ``Por acreditar''.}
    \UCpaso[\UCsist] Muestra el mensaje \cdtIdRef{MSG28}{Operación no permitida por estado de la entidad} en la pantalla en que se encuentre navegando el
    actor indicando que no puede consultar los indicadores debido a que la escuela no se encuentra en el estado ``Por acreditar''. 
 \end{UCtrayectoriaA}
